covid_data<-read.csv('covid19-in-india_1/covid_19_india.csv',stringsAsFactors = FALSE)

present_cases<-covid_data%>%filter(covid_data$Date==latest_date)
present_cases<-data.frame(State=present_cases$State.UnionTerritory,Total_cases=present_cases$Confirmed,stringsAsFactors = FALSE)

##Matching some of the state names in the "present_cases" with the names present in the shape file.

present_cases$State[present_cases$State=="Uttarakhand"]<-"Uttaranchal"
present_cases$State[present_cases$State=="Dadar Nagar Haveli"]<-"Dadra and Nagar Haveli"
present_cases$State[present_cases$State=="Telengana"]<-"Telangana"


##Reading the shape file and plotting.

shape_file<-st_read("F:/pdf e-reading/Mtech/2nd semester/Data visualisation lab/Temporal_chart/New folder/Shapefiles/Shapefiles/India.shp")

  dotsInPolys(map_india, as.integer(dots[, x]), f="random")
})

# then plot the dots

  plot(map_india,cex.main=0.8)+title("National Parks in India")
  plot(spdf[[i]],main="tt",add = T, pch = 19, cex = 0.8, col = pal[i])
}

## Selecting only India,China and USA airports

flights <- read.csv("http://www.stanford.edu/~cengel/cgi-bin/anthrospace/wp-content/uploads/2012/03/PEK-openflights-export-2012-03-19.csv", as.is=TRUE, header=TRUE)


airport_locations <- airports[, c("IATA","longitude", "latitude")]

##Frequency of flights between cities
flights_freq <- ddply(flights, c("From","To"), function(x) count(x$To))

location_origin <- left_join(flights_freq, airport_locations, by=c("From"="IATA") )
location_all <- left_join(location_origin, airport_locations, by=c("To"="IATA") )

ggplot()+geom_polygon(map_df,mapping=aes(long,lat,group=group,fill="red"))+geom_segment(location_all,mapping=aes(x=longitude.x,y=latitude.x,xend=longitude.y,yend=latitude.y))+ggtitle("Flights from Beijing ")+xlab("latitude")+ylab("longitude")
